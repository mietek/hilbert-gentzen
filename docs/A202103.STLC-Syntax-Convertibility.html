data _≅_ {Γ} : ∀ {A} → Γ ⊢ A → Γ ⊢ A → Set where

  -- Equivalence rules.

  refl  : ∀ {A} {t : Γ ⊢ A} →
          t ≅ t

  trans : ∀ {A} {t t′ t″ : Γ ⊢ A} →
          t ≅ t′ → t′ ≅ t″ → t ≅ t″

  sym   : ∀ {A} {t t′ : Γ ⊢ A} →
          t ≅ t′ → t′ ≅ t

  -- Congruence rules.

  cong-lam : ∀ {A B} {t t′ : Γ , A ⊢ B} →
             t ≅ t′ → lam t ≅ lam t′

  cong-app : ∀ {A B} {t₁ t₁′ : Γ ⊢ A ⊃ B} {t₂ t₂′ : Γ ⊢ A} →
             t₁ ≅ t₁′ → t₂ ≅ t₂′ → app t₁ t₂ ≅ app t₁′ t₂′

  -- Reduction and expansion rules.

  red-app-lam : ∀ {A B} (t₁ : Γ , A ⊢ B) (t₂ : Γ ⊢ A) →
                app (lam t₁) t₂ ≅ cut t₂ t₁

  exp-lam     : ∀ {A B} (t : Γ ⊢ A ⊃ B) →
                t ≅ lam (app (wk t) (var top))

≅←≡ : ∀ {Γ A} {t t′ : Γ ⊢ A} → t ≡ t′ → t ≅ t′
≅←≡ refl = refl
